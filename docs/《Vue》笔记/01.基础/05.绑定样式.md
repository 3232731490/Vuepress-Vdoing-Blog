## 绑定样式

```vue
	<style>
        .basic{width: 100%;height: 20px;}
        .happy{ background-color:aqua;}
        .sad{background-color:beige;}
        .normal{background-color:black;}
        .nilu1{background-color:chartreuse;}
        .nilu2{color:red;}
        .nilu3{border-radius: 5px;}
    </style>		

	<div>
		<!--绑定class样式--字符串写法，适用于： 样式的类名不确定，需要动态指定-->
        <h3 class="basic" :class='a' @click='change'>{{name}}</h3>
        <!--绑定class样式--数组写法，适用于： 要绑定的样式个数不确定，名字不确定-->
        <h3 class="basic" :class='arr' @click='change'>{{name}}</h3>
        <!--绑定class样式--对象写法，适用于： 要绑定的样式个数确定，名字确定,但要动态绑定是否使用-->
        <h3 class="basic" :class='obj' @click='change'>{{name}}</h3>

        <hr/>
        <!--绑定style样式--字符串写法-->
        <h3 class="basic" :style="{fontSize:fsize+'px'}"  @click='change'>{{name}}</h3>
        <hr/>
        <!--绑定style样式--对象写法-->
        <h3 class="basic" :style="styleObj"  @click='change'>{{name}}</h3>
        <!--也有数组写法，与class类似-->
	</div>

	<script>
		new Vue({
            el:'#root',
            data:{
				name:'111',
                a:'happy',
                arr:['nilu1','nilu2','nilu3'],
                obj:{
                    nilu1:true,
                    nilu2:false,
                    nilu3:true
                },
                fsize:40,
                styleObj:{
                    fontSize:2 + 'em'
                }
            },
            methods: {
                change(){
                    const arr = ['happy','sad','normal']
                    const num = Math.floor(Math.random()*3)
                    this.a=arr[num]
                }
            },
		})
	</script>
        
```
